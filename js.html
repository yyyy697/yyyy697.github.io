// 弹出对话框 alert("即将改变背景色并显示当前时间！"); // 改变屏幕背景色（随机颜色） document.body.style.backgroundColor = getRandomColor(); // 显示当前时间 function showCurrentTime() { const now = new Date(); const timeString = now.toLocaleTimeString(); const dateString = now.toLocaleDateString(); // 创建一个显示时间的div元素 const timeDisplay = document.createElement('div'); timeDisplay.id = 'timeDisplay'; timeDisplay.style.position = 'fixed'; timeDisplay.style.top = '20px'; timeDisplay.style.right = '20px'; timeDisplay.style.backgroundColor = 'white'; timeDisplay.style.padding = '10px'; timeDisplay.style.border = '1px solid black'; timeDisplay.style.borderRadius = '5px'; timeDisplay.style.zIndex = '1000'; timeDisplay.textContent = `当前日期: ${dateString}\n当前时间: ${timeString}`; document.body.appendChild(timeDisplay); // 每秒更新时间 setInterval(() => { const now = new Date(); timeDisplay.textContent = `当前日期: ${now.toLocaleDateString()}\n当前时间: ${now.toLocaleTimeString()}`; }, 1000); } // 生成随机颜色 function getRandomColor() { const letters = '0123456789ABCDEF'; let color = '#'; for (let i = 0; i < 6; i++) { color += letters[Math.floor(Math.random() * 16)]; } return color; } // 调用显示时间的函数 showCurrentTime();
